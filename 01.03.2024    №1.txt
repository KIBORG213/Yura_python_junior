# Список продуктів, в які я готовий вкладати гроші
інвестиції = {
    "технологічні стартапи": "можливість інвестицій від $100,000",
    "відновлювальна енергетика": "можливість інвестицій від $50,000",
    "медичні технології": "можливість інвестицій від $200,000",
    "електромобілі": "можливість інвестицій від $150,000",
    "сільське господарство": "можливість інвестицій від $75,000"
}

# Функція для перевірки продукту та відповіді
def перевірка_інвестицій(продукт):
    if продукт.lower() in інвестиції:
        return f"Ви можете отримати {інвестиції[продукт.lower()]}."
    else:
        return "Вибачте, цей продукт на даний момент не входить до мого списку інвестицій."

# Запит користувача та виведення відповіді
продукт = input("Введіть назву продукту: ")
відповідь = перевірка_інвестицій(продукт)
print(відповідь)

# Список завдань для кожного числа на гральному кубику
завдання = {
    1: "Скажіть щось позитивне про себе.",
    2: "Розкажіть анекдот.",
    3: "Назвіть трьох своїх улюблених акторів або актрис.",
    4: "Розповісти про свої плани на наступний тиждень.",
    5: "Поділіться найсмішнішим досвідом зі свого життя.",
    6: "Відповісти на питання: 'Якби ви могли побачити будь-якого живого або мертвого, кого б ви обрали?'"
}

# Функція для кидання грального кубика
def кинути_кубик():
    return random.randint(1, 6)

# Кидання кубика та виконання завдання
результат = кинути_кубик()
завдання_гравця = завдання[результат]
print(f"Вам випало число {результат}!")
print(f"Ваше завдання: {завдання_гравця}")
